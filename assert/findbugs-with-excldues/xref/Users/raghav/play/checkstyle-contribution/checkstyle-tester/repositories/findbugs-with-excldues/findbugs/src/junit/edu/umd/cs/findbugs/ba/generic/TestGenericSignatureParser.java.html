20  package edu.umd.cs.findbugs.ba.generic;
21  
22  import java.util.Iterator;
23  
24  import junit.framework.TestCase;
25  
26  /**
27   * @author Nat Ayewah
28   */
29  public class TestGenericSignatureParser extends TestCase {
30  
31      public void processTest(String genericSignature, String... substrings) {
32          GenericSignatureParser parser = new GenericSignatureParser(genericSignature);
33          Iterator<String> iter = parser.parameterSignatureIterator();
34  
35          for (String s : substrings) {
36              assertTrue(iter.hasNext());
37              assertEquals(s, iter.next());
38          }
39          assertFalse(iter.hasNext());
40      }
41  
42      public void testSignatures() {
43          processTest("(Ljava/lang/Comparable;)V", "Ljava/lang/Comparable;");
44  
45          processTest("(Ljava/lang/Comparable;TE;**[Ljava/lang/Comparable;)V", "Ljava/lang/Comparable;", "TE;", "*", "*",
46                  "[Ljava/lang/Comparable;");
47  
48          processTest("(TE;*+[Ljava/lang/Comparable;-TV;)V", "TE;", "*", "+[Ljava/lang/Comparable;", "-TV;");
49      }
50  }
