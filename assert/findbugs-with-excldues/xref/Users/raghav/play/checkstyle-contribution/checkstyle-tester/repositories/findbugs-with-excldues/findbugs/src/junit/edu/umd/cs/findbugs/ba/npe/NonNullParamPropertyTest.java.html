1   package edu.umd.cs.findbugs.ba.npe;
2   
3   import junit.framework.Assert;
4   import junit.framework.TestCase;
5   import edu.umd.cs.findbugs.ba.interproc.ParameterProperty;
6   
7   public class NonNullParamPropertyTest extends TestCase {
8   
9       ParameterProperty empty;
10  
11      ParameterProperty nonEmpty;
12  
13      ParameterProperty extremes;
14  
15      @Override
16      protected void setUp() throws Exception {
17          empty = new ParameterProperty();
18  
19          nonEmpty = new ParameterProperty();
20          nonEmpty.setParamWithProperty(11, true);
21          nonEmpty.setParamWithProperty(25, true);
22  
23          extremes = new ParameterProperty();
24          extremes.setParamWithProperty(0, true);
25          extremes.setParamWithProperty(31, true);
26      }
27  
28      public void testEmpty() {
29          for (int i = 0; i < 32; ++i) {
30              Assert.assertFalse(empty.hasProperty(i));
31          }
32      }
33  
34      public void testIsEmpty() {
35          Assert.assertTrue(empty.isEmpty());
36          Assert.assertFalse(nonEmpty.isEmpty());
37          Assert.assertFalse(extremes.isEmpty());
38      }
39  
40      public void testNonEmpty() {
41          Assert.assertTrue(nonEmpty.hasProperty(11));
42          Assert.assertTrue(nonEmpty.hasProperty(25));
43          Assert.assertFalse(nonEmpty.hasProperty(5));
44      }
45  
46      public void testExtremes() {
47          Assert.assertTrue(extremes.hasProperty(0));
48          Assert.assertTrue(extremes.hasProperty(31));
49          Assert.assertFalse(extremes.hasProperty(10));
50      }
51  
52      public void testOutOfBounds() {
53          Assert.assertFalse(nonEmpty.hasProperty(-1));
54          Assert.assertFalse(nonEmpty.hasProperty(32));
55      }
56  }
