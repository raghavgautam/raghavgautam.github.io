7   import static org.junit.Assert.assertTrue;

9   import java.io.File;
10  import java.net.URL;
11  
12  import net.sourceforge.pmd.TestBase;
13  
14  import org.apache.commons.io.IOUtils;
15  import org.junit.Test;
16  
17  /**
18   * @author rpelisse
19   *
20   */
21  public class PmdBuildTaskTest extends TestBase {
22  
23      @Test
24      public void antTask() throws Exception {
25          PmdBuildTask task = new PmdBuildTask();
26          task.setRulesDirectory(TEST_DIR + "src/main/resources/rulesets");
27          task.setTarget(TEST_DIR + "target");
28          task.setSiteXml(TEST_DIR + "src/site/site.pre.xml");
29          task.setSiteXmlTarget(TEST_DIR + "src/site/site.xml");
30          task.setRuntimeClasspath(new URL[] { new File("target/test-classes").toURI().toURL() });
31          task.execute();
32  
33          String site = IOUtils.toString(new File(TEST_DIR + "src/site/site.xml").toURI());
34          assertTrue(site.contains("<item name=\"Basic\""));
35          assertTrue(site.contains("<item name=\"Code Size\""));
36          assertTrue(site.indexOf("<item name=\"Basic\"") < site.indexOf("<item name=\"Code Size\""));
37      }
38  }
