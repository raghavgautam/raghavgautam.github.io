17  /**
18   * Tests rehashing with distributed caches with L1 enabled.
19   *
20   * @author Galder Zamarre√±o
21   * @since 5.2
22   */
23  @Test(groups = "functional", testName = "distribution.rehash.RehashWithL1Test")
24  public class RehashWithL1Test extends MultipleCacheManagersTest {
25  
26     ConfigurationBuilder builder;
27  
28     @Override
29     protected void createCacheManagers() throws Throwable {
30        builder = getDefaultClusteredCacheConfig(CacheMode.DIST_SYNC, false);
31        // Enable rehashing explicitly
32        builder.clustering().stateTransfer().fetchInMemoryState(true)
33              .hash().l1().enable();
34        createClusteredCaches(2, builder);
35     }
36  
37     public void testPutWithRehashAndCacheClear() throws Exception {
38        Future<Void> node3Join = null;
39        int opCount = 100;
40        for (int i = 0; i < opCount; i++) {
41           cache(0).put("k" + i, "some data");
42           if (i == (opCount / 2)) {
43              node3Join = fork(new Callable<Void>() {
44                 @Override
45                 public Void call() throws Exception {
46                    EmbeddedCacheManager cm = addClusterEnabledCacheManager(builder,
47                          new TransportFlags().withMerge(true));
48                    cm.getCache();
49                    return null;
50                 }
51              });
52           }
53           Thread.sleep(10);
54        }
55  
56        if (node3Join == null) throw new Exception("Node 3 not joined");
57        node3Join.get();
58  
59        for (int i = 0; i < opCount; i++) {
60           cache(0).remove("k" + i);
61           Thread.sleep(10);
62        }
63  
64        for (int i = 0; i < opCount; i++) {
65           String key = "k" + i;
66           assertFalse(cache(0).containsKey(key));
67           assertFalse("Key: " + key + " is present in cache at " + cache(1),
68                 cache(1).containsKey(key));
69           assertFalse("Key: " + key + " is present in cache at " + cache(2),
70                 cache(2).containsKey(key));
71        }
72  
73        assertEquals(0, cache(0).size());
74        assertEquals(0, cache(1).size());
75        assertEquals(0, cache(2).size());
76     }
