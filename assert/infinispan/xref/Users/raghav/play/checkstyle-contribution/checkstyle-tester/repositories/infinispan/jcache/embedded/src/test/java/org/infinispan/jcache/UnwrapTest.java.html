17  
18  @Test(groups = "functional", testName = "jcache.UnwrapTest")
19  public class UnwrapTest {
20  
21     public void testUnwrap() {
22        withCacheManager(new CacheManagerCallable(
23              TestCacheManagerFactory.createCacheManager(false)) {
24           @Override
25           public void call() {
26              cm.defineConfiguration("UnwrapCache", new ConfigurationBuilder().build());
27              CacheManager jCacheManager = new JCacheManager(URI.create("UnwrapCacheManager"), cm, null);
28              Cache<Object, Object> jcache = jCacheManager.getCache("UnwrapCache");
29              assertTrue(jCacheManager.unwrap(JCacheManager.class) != null);
30              assertTrue(jcache.unwrap(JCache.class) != null);
31              assertTrue(jCacheManager.unwrap(EmbeddedCacheManager.class) != null);
32              assertTrue(jcache.unwrap(org.infinispan.Cache.class) != null);
33           }
34        });
35     }
