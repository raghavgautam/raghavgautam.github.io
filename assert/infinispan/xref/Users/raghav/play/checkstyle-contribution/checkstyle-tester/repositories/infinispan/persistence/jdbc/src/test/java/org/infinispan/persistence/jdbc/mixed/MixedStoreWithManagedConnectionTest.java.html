23  @Test(groups = "functional", testName = "persistence.jdbc.mixed.MixedStoreWithManagedConnectionTest")
24  public class MixedStoreWithManagedConnectionTest extends ManagedConnectionFactoryTest {
25  
26     @Override
27     protected AdvancedLoadWriteStore createStore() throws Exception {
28  
29        ConfigurationBuilder builder = TestCacheManagerFactory.getDefaultCacheConfiguration(false);
30        JdbcMixedStoreConfigurationBuilder storeBuilder = builder
31              .persistence()
32              .addStore(JdbcMixedStoreConfigurationBuilder.class);
33        storeBuilder.dataSource().jndiUrl(getDatasourceLocation());
34        UnitTestDatabaseManager.setDialect(storeBuilder);
35        UnitTestDatabaseManager.buildTableManipulation(storeBuilder.stringTable(), false);
36        UnitTestDatabaseManager.buildTableManipulation(storeBuilder.binaryTable(), true);
37  
38        storeBuilder
39              .binaryTable()
40              .tableNamePrefix("BINARY_TABLE")
41              .stringTable()
42              .tableNamePrefix("STRINGS_TABLE");
43  
44        JdbcMixedStore jdbcMixed = new JdbcMixedStore();
45        jdbcMixed.init(createContext(builder.build()));
46        return jdbcMixed;
47     }
48  
49     public void testLoadFromFile() throws Exception {
50        TestingUtil.withCacheManager(new CacheManagerCallable(TestCacheManagerFactory.fromXml("configs/managed/mixed-managed-connection-factory.xml"), true) {
51           @Override
52           public void call() {
53              Cache<String, String> first = cm.getCache("first");
54              Cache<String, String> second = cm.getCache("second");
55  
56              StoreConfiguration firstCacheLoaderConfig = first.getCacheConfiguration().persistence().stores().get(0);
57              assertNotNull(firstCacheLoaderConfig);
58              StoreConfiguration secondCacheLoaderConfig = second.getCacheConfiguration().persistence().stores().get(0);
59              assertNotNull(secondCacheLoaderConfig);
60              assertTrue(firstCacheLoaderConfig instanceof JdbcMixedStoreConfiguration);
61              assertTrue(secondCacheLoaderConfig instanceof JdbcMixedStoreConfiguration);
62              JdbcMixedStore loader = (JdbcMixedStore) TestingUtil.getFirstLoader(first);
63              assertTrue(loader.getConnectionFactory() instanceof ManagedConnectionFactory);
64           }
65        });
66     }
67  
68     @Override
69     public String getDatasourceLocation() {
70        return "java:/MixedStoreWithManagedConnectionTest/DS";
71     }
72  
73     @Override
74     protected boolean storePurgesAllExpired() {
75        return false;
76     }
