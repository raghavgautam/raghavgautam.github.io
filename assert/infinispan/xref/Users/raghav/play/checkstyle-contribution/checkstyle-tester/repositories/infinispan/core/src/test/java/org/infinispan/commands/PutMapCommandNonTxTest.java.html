1   package org.infinispan.commands;
2   
3   import org.infinispan.configuration.cache.CacheMode;
4   import org.infinispan.configuration.cache.ConfigurationBuilder;
5   import org.infinispan.context.Flag;
6   import org.infinispan.distribution.MagicKey;
7   import org.infinispan.test.MultipleCacheManagersTest;
8   import org.infinispan.test.fwk.CleanupAfterMethod;
9   import org.testng.annotations.Test;
10  
11  import java.util.Collections;
12  import java.util.concurrent.Future;
13  
14  import static org.junit.Assert.*;
15  
16  @Test(groups = "functional", testName = "commands.PutMapCommandNonTxTest")
17  @CleanupAfterMethod
18  public class PutMapCommandNonTxTest extends MultipleCacheManagersTest {
19  
20     @Override
21     protected void createCacheManagers() {
22        ConfigurationBuilder dcc = getDefaultClusteredCacheConfig(CacheMode.DIST_SYNC, false);
23        dcc.clustering().hash().numOwners(3).l1().disable();
24  
25        createCluster(dcc, 3);
26        waitForClusterToForm();
27     }
28  
29     public void testPutMapCommandSyncOnPrimaryOwner() throws Exception {
30        testPutMapCommand(true, true);
31     }
32  
33     public void testPutMapCommandAsyncOnPrimaryOwner() throws Exception {
34        testPutMapCommand(false, true);
35     }
36  
37     public void testPutMapCommandSyncOnBackupOwner() throws Exception {
38        testPutMapCommand(true, false);
39     }
40  
41     public void testPutMapCommandAsyncOnBackupOwner() throws Exception {
42        testPutMapCommand(false, false);
43     }
44  
45     private void testPutMapCommand(boolean sync, boolean putOnPrimary) throws Exception {
46        MagicKey key = new MagicKey("key", cache(0));
47  
48        if (sync) {
49           cache(putOnPrimary ? 0 : 1).putAll(Collections.singletonMap(key, "value"));
50        } else {
51           Future<Void> f = cache(putOnPrimary ? 0 : 1).putAllAsync(Collections.singletonMap(key, "value"));
52           assertNotNull(f);
53           assertNull(f.get());
54           assertTrue(f.isDone());
55           assertFalse(f.isCancelled());
56        }
57  
58        assertEquals("value", cache(0).getAdvancedCache().withFlags(Flag.SKIP_REMOTE_LOOKUP).get(key));
59        assertEquals("value", cache(1).getAdvancedCache().withFlags(Flag.SKIP_REMOTE_LOOKUP).get(key));
60        assertEquals("value", cache(2).getAdvancedCache().withFlags(Flag.SKIP_REMOTE_LOOKUP).get(key));
61     }
62  }
