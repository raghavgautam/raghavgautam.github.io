20  @Test(groups = "unit", testName = "distribution.topologyaware.TopologyAwareSyncConsistentHashFactoryTest")
21  public class TopologyAwareSyncConsistentHashFactoryTest extends TopologyAwareConsistentHashFactoryTest {
22  
23     private Log log = LogFactory.getLog(TopologyAwareSyncConsistentHashFactoryTest.class);
24  
25     public TopologyAwareSyncConsistentHashFactoryTest() {
26        // Increase the number of segments to eliminate collisions (which would cause extra segment movements,
27        // causing testConsistencyAfterLeave to fail.)
28        numSegments = 1000;
29     }
30  
31     @Override
32     protected ConsistentHashFactory<DefaultConsistentHash> createConsistentHashFactory() {
33        return new TopologyAwareSyncConsistentHashFactory();
34     }
35  
36     @Override
37     protected void assertDistribution(int numOwners, List<Address> currentMembers) {
38        TopologyAwareOwnershipStatistics stats = new TopologyAwareOwnershipStatistics(ch);
39        log.tracef("Ownership stats: " + stats);
40        for (Address node : currentMembers) {
41           int maxPrimarySegments = stats.computeExpectedSegments(numSegments, 1, node);
42           int maxSegments = stats.computeExpectedSegments(numSegments, numOwners, node);
43           assertTrue(0.65 * maxPrimarySegments <= stats.getPrimaryOwned(node));
44           assertTrue(stats.getPrimaryOwned(node) <= 1.2 * maxPrimarySegments);
45           assertTrue(0.65 * maxSegments <= stats.getOwned(node));
46           assertTrue(stats.getOwned(node) <= 1.2 * maxSegments);
47        }
48     }
49  }
