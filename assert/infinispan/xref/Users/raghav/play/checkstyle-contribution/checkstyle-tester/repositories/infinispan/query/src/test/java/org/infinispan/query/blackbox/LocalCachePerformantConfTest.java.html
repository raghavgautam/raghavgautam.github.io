1   package org.infinispan.query.blackbox;
2   
3   import org.infinispan.manager.EmbeddedCacheManager;
4   import org.infinispan.test.TestingUtil;
5   import org.infinispan.test.fwk.TestCacheManagerFactory;
6   import org.testng.annotations.Test;
7   
8   import java.io.File;
9   
10  import static org.junit.Assert.assertTrue;
11  
12  /**
13   * Tests the functionality of the queries in case when the NRT index manager is used in combination with FileStore.
14   *
15   * @author Anna Manukyan
16   */
17  @Test(groups = "functional", testName = "query.blackbox.LocalCachePerformantConfTest")
18  public class LocalCachePerformantConfTest extends LocalCacheTest {
19  
20     /** the file constant needs to match what's defined in the configuration file **/
21     private final String indexDirectory = System.getProperty("java.io.tmpdir") + File.separator + "LocalCachePerformantConfTest";
22  
23     @Override
24     protected EmbeddedCacheManager createCacheManager() throws Exception {
25        cacheManager = TestCacheManagerFactory.fromXml("testconfig-LocalCachePerformantConfTest.xml");
26        cache = cacheManager.getCache("Indexed");
27  
28        return cacheManager;
29     }
30  
31     @Override
32     protected void setup() throws Exception {
33        TestingUtil.recursiveFileRemove(indexDirectory);
34        boolean created = new File(indexDirectory).mkdirs();
35        assertTrue(created);
36        super.setup();
37     }
38  
39     @Override
40     protected void teardown() {
41        try {
42           super.teardown();
43        } finally {
44           TestingUtil.recursiveFileRemove(indexDirectory);
45        }
46     }
47  }
