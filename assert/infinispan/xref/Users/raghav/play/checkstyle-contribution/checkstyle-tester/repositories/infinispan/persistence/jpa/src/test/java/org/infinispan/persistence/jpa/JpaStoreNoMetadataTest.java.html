59     // Without metadata we cannot purge anything - we should test this expected behaviour
60     @Override
61     public void testPurgeExpired() throws Exception {
62        long lifespan = 6000;
63        long idle = 4000;
64        InternalCacheEntry ice1 = TestInternalCacheEntryFactory.create("k1", wrap("k1", "v1"), lifespan);
65        cl.write(TestingUtil.marshalledEntry(ice1, getMarshaller()));
66        InternalCacheEntry ice2 = TestInternalCacheEntryFactory.create("k2", wrap("k2", "v2"), -1, idle);
67        cl.write(TestingUtil.marshalledEntry(ice2, getMarshaller()));
68        InternalCacheEntry ice3 = TestInternalCacheEntryFactory.create("k3", wrap("k3", "v3"), lifespan, idle);
69        cl.write(TestingUtil.marshalledEntry(ice3, getMarshaller()));
70        InternalCacheEntry ice4 = TestInternalCacheEntryFactory.create("k4", wrap("k4", "v4"), -1, -1);
71        cl.write(TestingUtil.marshalledEntry(ice4, getMarshaller())); // immortal entry
72        InternalCacheEntry ice5 = TestInternalCacheEntryFactory.create("k5", wrap("k5", "v5"), lifespan * 1000, idle * 1000);
73        cl.write(TestingUtil.marshalledEntry(ice5, getMarshaller())); // long life mortal entry
74        assertTrue(cl.contains("k1"));
75        assertTrue(cl.contains("k2"));
76        assertTrue(cl.contains("k3"));
77        assertTrue(cl.contains("k4"));
78        assertTrue(cl.contains("k5"));
79  
80        purgeExpired();
81  
82        assertTrue(cl.contains("k1"));
83        assertTrue(cl.contains("k2"));
84        assertTrue(cl.contains("k3"));
85        assertTrue(cl.contains("k4"));
86        assertTrue(cl.contains("k5"));
87     }
88  
89     @Override
90     public void testStopStartDoesNotNukeValues() throws InterruptedException, PersistenceException {
91        // Without metadata the entries do not expire - we should test this expected behaviour
92        assertFalse(cl.contains("k1"));
93        assertFalse(cl.contains("k2"));
94  
95        long lifespan = 1;
96        long idle = 1;
97        InternalCacheEntry se1 = TestInternalCacheEntryFactory.create("k1", wrap("k1", "v1"), lifespan);
98        InternalCacheEntry se2 = TestInternalCacheEntryFactory.create("k2", wrap("k2", "v2"));
99        InternalCacheEntry se3 = TestInternalCacheEntryFactory.create("k3", wrap("k3", "v3"), -1, idle);
100       InternalCacheEntry se4 = TestInternalCacheEntryFactory.create("k4", wrap("k4", "v4"), lifespan, idle);
101 
102       cl.write(TestingUtil.marshalledEntry(se1, getMarshaller()));
103       cl.write(TestingUtil.marshalledEntry(se2, getMarshaller()));
104       cl.write(TestingUtil.marshalledEntry(se3, getMarshaller()));
105       cl.write(TestingUtil.marshalledEntry(se4, getMarshaller()));
106 
107       timeService.advance(lifespan + 1);
108 
109       cl.stop();
110       cl.start();
111       assertTrue(se1.isExpired(System.currentTimeMillis()));
112       assertTrue(cl.contains("k1"));
113       assertEquals("v1", unwrap(cl.load("k1").getValue()));
114       assertTrue(cl.contains("k2"));
115       assertEquals("v2", unwrap(cl.load("k2").getValue()));
116       assertTrue(se3.isExpired(System.currentTimeMillis()));
117       assertTrue(cl.contains("k3"));
118       assertEquals("v3", unwrap(cl.load("k3").getValue()));
119       assertTrue(se3.isExpired(System.currentTimeMillis()));
120       assertTrue(cl.contains("k4"));
121       assertEquals("v4", unwrap(cl.load("k4").getValue()));
122    }
