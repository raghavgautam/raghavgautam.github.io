19  package com.opensymphony.xwork2.util.profiling;
20  
21  import junit.framework.TestCase;
22  
23  /**
24   * @author tmjee
25   * @version $Date$ $Id$
26   */
27  public class UtilTimerStackTest extends TestCase {
28  
29      protected String activateProp;
30      protected String minTimeProp;
31  
32  
33      public void testActivateInactivate() throws Exception {
34          UtilTimerStack.setActive(true);
35          assertTrue(UtilTimerStack.isActive());
36          UtilTimerStack.setActive(false);
37          assertFalse(UtilTimerStack.isActive());
38      }
39  
40  
41      public void testPushPop() throws Exception {
42          UtilTimerStack.push("p1");
43          Thread.sleep(1050);
44          ProfilingTimerBean bean = UtilTimerStack.current.get();
45          assertTrue(bean.startTime > 0);
46          UtilTimerStack.pop("p1");
47          assertTrue(bean.totalTime > 1000);
48      }
49  
50  
51      public void testProfileCallback() throws Exception {
52  
53          MockProfilingBlock<String> block = new MockProfilingBlock<String>() {
54              @Override
55              public String performProfiling() throws Exception {
56                  Thread.sleep(1050);
57                  return "OK";
58              }
59          };
60          String result = UtilTimerStack.profile("p1", block);
61          assertEquals(result, "OK");
62          assertNotNull(block.getProfilingTimerBean());
63          assertTrue(block.getProfilingTimerBean().totalTime >= 1000);
64  
65      }
66  
67  
68      public void testProfileCallbackThrowsException() throws Exception {
69          try {
70              UtilTimerStack.profile("p1",
71                      new UtilTimerStack.ProfilingBlock<String>() {
72                          public String doProfiling() throws Exception {
73                              throw new RuntimeException("test");
74                          }
75                      });
76              fail("exception should have been thrown");
77          }
78          catch (Exception e) {
79              assertTrue(true);
80          }
81      }
82  
83  
84      @Override
85      protected void setUp() throws Exception {
86          super.setUp();
87  
88          activateProp = System.getProperty(UtilTimerStack.ACTIVATE_PROPERTY);
89          minTimeProp = System.getProperty(UtilTimerStack.MIN_TIME);
90  
91          System.setProperty(UtilTimerStack.ACTIVATE_PROPERTY, "true");
92          UtilTimerStack.setActive(true);
93          System.setProperty(UtilTimerStack.MIN_TIME, "0");
94      }
95  
96  
97      @Override
98      protected void tearDown() throws Exception {
99  
100         if (activateProp != null) {
101             System.setProperty(UtilTimerStack.ACTIVATE_PROPERTY, activateProp);
102         } else {
103             System.clearProperty(UtilTimerStack.ACTIVATE_PROPERTY);
104         }
105         if (minTimeProp != null) {
106             System.setProperty(UtilTimerStack.MIN_TIME, minTimeProp);
107         } else {
108             System.clearProperty(UtilTimerStack.ACTIVATE_PROPERTY);
109         }
110 
111 
112         activateProp = null;
113         minTimeProp = null;
114 
115         super.tearDown();
116     }
117 
118 
119     public abstract class MockProfilingBlock<T> implements UtilTimerStack.ProfilingBlock<T> {
120 
121         private ProfilingTimerBean bean;
122 
123         public T doProfiling() throws Exception {
124             bean = UtilTimerStack.current.get();
125             return performProfiling();
126         }
127 
128         public ProfilingTimerBean getProfilingTimerBean() {
129             return bean;
130         }
131 
132         public abstract T performProfiling() throws Exception;
133     }
134 }
