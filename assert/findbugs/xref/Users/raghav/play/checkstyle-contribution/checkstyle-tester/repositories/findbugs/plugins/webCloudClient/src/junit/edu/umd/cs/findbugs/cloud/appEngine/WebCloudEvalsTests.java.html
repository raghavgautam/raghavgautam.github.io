432     private static Evaluation createEvaluation(String designation, long when, String comment, String who) {
433         return Evaluation.newBuilder().setWhen(when).setDesignation(designation).setComment(comment).setWho(who).build();
434     }
435 
436     private static Issue createIssueToReturn(Evaluation... evaluations) {
437         return Issue.newBuilder().setFirstSeen(SAMPLE_DATE + 100).setLastSeen(SAMPLE_DATE + 300)
438                 .addAllEvaluations(Arrays.asList(evaluations)).build();
439     }
440 
441     private static Issue fillMissingFields(Issue prototype, BugInstance source, Evaluation... evalsToAdd) {
442         return Issue.newBuilder(prototype).setBugPattern(source.getAbbrev())
443                 .setHash(WebCloudProtoUtil.encodeHash(source.getInstanceHash()))
444                 .setPrimaryClass(source.getPrimaryClass().getClassName()).setPriority(1)
445                 .addAllEvaluations(Arrays.asList(evalsToAdd)).build();
446     }
447 
448     private static void checkStoredEvaluationMatches(Evaluation expectedEval, BugDesignation designation) {
449         assertNotNull("Did not get any designation", designation);
450         assertEquals(expectedEval.getComment(), designation.getAnnotationText());
451         assertEquals(expectedEval.getDesignation(), designation.getDesignationKey());
452         assertEquals(expectedEval.getWho(), designation.getUser());
453         assertEquals(expectedEval.getWhen(), designation.getTimestamp());
454     }
455 
456     private static void checkUploadedEvaluationMatches(BugInstance expectedValues, UploadEvaluation uploadMsg) {
457         assertEquals(555, uploadMsg.getSessionId());
458         assertEquals(expectedValues.getInstanceHash(), WebCloudProtoUtil.decodeHash(uploadMsg.getHash()));
459         assertEquals(expectedValues.getUserDesignationKey(), uploadMsg.getEvaluation().getDesignation());
460         assertEquals(expectedValues.getAnnotationText(), uploadMsg.getEvaluation().getComment());
461     }
