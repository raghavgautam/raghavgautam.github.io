20  package edu.umd.cs.findbugs;
21  
22  import java.io.IOException;
23  import java.util.logging.Logger;
24  
25  import org.dom4j.DocumentException;
26  
27  import edu.umd.cs.findbugs.annotations.SuppressFBWarnings;
28  
29  /**
30   * Add an annotation string to every BugInstance in a BugCollection.
31   */
32  public class AddAnnotation {
33      private static final Logger LOGGER = Logger.getLogger(AddAnnotation.class.getName());
34  
35      private final BugCollection bugCollection;
36  
37      private final String annotation;
38  
39      public AddAnnotation(BugCollection bugCollection, String annotation) {
40          this.bugCollection = bugCollection;
41          this.annotation = annotation;
42      }
43  
44      public AddAnnotation(String resultsFile, String annotation) throws IOException, DocumentException {
45          this(new SortedBugCollection(), annotation);
46          bugCollection.readXML(resultsFile);
47      }
48  
49      public BugCollection getBugCollection() {
50          return bugCollection;
51      }
52  
53      public Project getProject() {
54          return bugCollection.getProject();
55      }
56  
57      public void execute() {
58          for (BugInstance bugInstance : bugCollection) {
59              // Don't add the annotation if it is already present
60              if (bugInstance.annotationTextContainsWord(this.annotation)) {
61                  continue;
62              }
63  
64              String annotation = bugInstance.getAnnotationText();
65              StringBuilder buf = new StringBuilder();
66              if (!"".equals(annotation)) {
67                  buf.append(annotation);
68                  buf.append('\n');
69              }
70              buf.append(this.annotation);
71              bugInstance.setAnnotationText(buf.toString(), bugCollection);
72          }
73      }
74  
75      @SuppressFBWarnings("DM_EXIT")
76      public static void main(String[] argv) throws Exception {
77          if (argv.length != 2) {
78              System.err.println("Usage: " + AddAnnotation.class.getName() + " <results file> <annotation>");
79              System.exit(1);
80          }
81  
82          String filename = argv[0];
83          String annotation = argv[1];
84  
85          AddAnnotation addAnnotation = new AddAnnotation(filename, annotation);
86          addAnnotation.execute();
87  
88          addAnnotation.getBugCollection().writeXML(filename);
89      }
90  }
